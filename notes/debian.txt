# 用 n 升级nodejs
	npm i -g n
	# 查看系统node的安装路径
	which node
	# 通过N_PREFIX变量来修改 n 的默认node安装路径
	vim ~/.bash_profile 
	# 添加以下2行
	export N_PREFIX=/urs/bin/node #node实际安装位置
	export PATH=$N_PREFIX/bin:$PATH
	# 执行source使修改生效
	source ~/.bash_profile
	echo $N_PREFIX
	# 升级
	n latest  #安装node最新版本
	n stable  #稳定版
	n v7.10.0 #指定版本
	n lts	  #LTS版本
